            <input type="range" id="mass" min="0.1" max="10" step="0.1" value="2.0">

            <input type="range" id="speed" min="1" max="100" value="10">

                <h4>實時力學數據</h4>

                <p>扭矩: <span id="torsion">0</span> N-mm</p>

                <p>衝擊力: <span id="impact">0</span> N</p>

    <script>

        // 這裡結合了你 App.tsx 的邏輯

        const hand = document.getElementById('hand');

        const massInput = document.getElementById('mass');

        const speedInput = document.getElementById('speed');

        let angle = 0;


        function update() {

            const m = parseFloat(massInput.value);

            const s = parseFloat(speedInput.value);

            

            document.getElementById('mVal').innerText = m;

            document.getElementById('sVal').innerText = s;


            // 動畫

            angle = (angle + s/10) % 360;

            hand.style.transform = `rotate(${angle}deg)`;


            // 計算

            const t = (m * 9.8 * 150 * (s/100)).toFixed(2);

            const i = (m * 15.5).toFixed(2);

            document.getElementById('torsion').innerText = t;

            document.getElementById('impact').innerText = i;


            const status = document.getElementById('status');

            if (t > 40) {

                status.innerText = "🚨 警告：軸心負荷過重";

                status.className = "warning";

            } else {

                status.innerText = "✅ 結構安全";

                status.className = "";

            }

            requestAnimationFrame(update);

        }

        update();

    </script>
